  DATABASE CHANGES
================================================================================

NEW COLUMN (users table):
  role VARCHAR(50) DEFAULT 'user'

NEW TABLE (approval_logs):
  id UUID PRIMARY KEY
  user_id UUID (references users)
  admin_id UUID (references users)
  status VARCHAR(50) -- 'approved' or 'rejected'
  notes TEXT
  created_at TIMESTAMP
  updated_at TIMESTAMP

NEW INDEXES:
  idx_users_role
  idx_users_approval_status
  idx_approval_logs_user_id
  idx_approval_logs_admin_id

================================================================================
  API ENDPOINTS
================================================================================

ADMIN AUTHENTICATION:
  POST /api/admin/login
  Input: { email, password }
  Output: { user, message }
  Auth: Required

USER MANAGEMENT:
  GET /api/admin/users
  Output: [{ id, name, email, status, ... }]
  Auth: Admin only

USER APPROVAL:
  PATCH /api/admin/users/[id]/approve
  Output: { user, message }
  Auth: Admin only

USER REJECTION:
  PATCH /api/admin/users/[id]/reject
  Input: { notes }
  Output: { user, message }
  Auth: Admin only

ROLE CHECK:
  GET /api/users/role?email=user@email.com
  Output: { role: 'user' | 'admin' }
  Auth: Public
